##############################################################################
# CMake
##############################################################################

cmake_minimum_required(VERSION 2.8.3)
project(concert_conductor)

##############################################################################
# Catkin
##############################################################################

find_package(catkin REQUIRED)
catkin_package()

catkin_python_setup()

##############################################################################
# Installs
##############################################################################

install(
    PROGRAMS 
        scripts/conductor.py
        scripts/concert_conductor_graph
    DESTINATION ${CATKIN_PACKAGE_BIN_DESTINATION}
    )
